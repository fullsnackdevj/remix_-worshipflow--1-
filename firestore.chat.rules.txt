// Add these rules to your existing Firestore Security Rules
// Firebase Console → Firestore Database → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── existing rules stay here ──────────────────────────────────────────────

    // ── Team Chat: allow authenticated users to read & write ─────────────────
    match /chat_channels/{channelId}/messages/{messageId} {
      // Any signed-in user can read messages
      allow read: if request.auth != null;

      // Any signed-in user can create their own messages
      allow create: if request.auth != null
                    && request.resource.data.userId == request.auth.uid;

      // Users can delete their own messages; admins can delete any
      allow delete: if request.auth != null
                    && resource.data.userId == request.auth.uid;

      // No client-side edits (messages are immutable once sent)
      allow update: if false;
    }
  }
}
